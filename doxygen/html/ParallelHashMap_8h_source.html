650 template <class K, class V>
651 void ParallelHashMap<K,V>::resize() {
652 
653  /* acquire all locks in order */
654  for (size_t i=0; i<_num_locks; i++)
655  omp_set_lock(&_locks[i]);
656 
657  /* recheck if resize needed */
658  if (2*_table->size() < _table->bucket_count()) {
659  /* release locks */
660  for (size_t i=0; i<_num_locks; i++)
661  omp_unset_lock(&_locks[i]);
662 
663  return;
664  }
665 
666  /* allocate new hash map of double the size */
667  FixedHashMap<K,V> *new_map =
668  new FixedHashMap<K,V>(2*_table->bucket_count());
669 
670  /* get keys, values, and number of elements */
671  K *key_list = _table->keys();
672  V *value_list = _table->values();
673 
674  /* insert key/value pairs into new hash map */
675  for (size_t i=0; i<_table->size(); i++)
676  new_map->insert(key_list[i], value_list[i]);
677 
678  /* save pointer of old table */
679  FixedHashMap<K,V> *old_table = _table;
680 
681  /* reassign pointer */
682  _table = new_map;
683 #pragma omp flush(_table)
684 
685  /* release all locks */
686  for (size_t i=0; i<_num_locks; i++)
687  omp_unset_lock(&_locks[i]);
688 
689  /* delete key and value list */
690  delete [] key_list;
691  delete [] value_list;
692 
693  /* wait for all threads to stop reading from the old table */
694  for (size_t i=0; i<_num_threads; i++)
695  while (_announce[i].value == old_table)
696  continue;
697 
698  /* free memory associated with old table */
699  delete old_table;
700 }
