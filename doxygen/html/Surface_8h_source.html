366 inline double Surface::getMinDistance(Point* point, double angle) {
367 
368  /* Point array for intersections with this Surface */
369  Point intersections[2];
370 
371  /* Find the intersection Point(s) */
372  int num_inters = this->intersection(point, angle, intersections);
373  double distance = INFINITY;
374 
375  /* If there is one intersection Point */
376  if (num_inters == 1)
377  distance = intersections[0].distanceToPoint(point);
378 
379  /* If there are two intersection Points */
380  else if (num_inters == 2) {
381  double dist1 = intersections[0].distanceToPoint(point);
382  double dist2 = intersections[1].distanceToPoint(point);
383 
384  /* Determine which intersection Point is nearest */
385  if (dist1 < dist2)
386  distance = dist1;
387  else
388  distance = dist2;
389  }
390 
391  return distance;
392 }
