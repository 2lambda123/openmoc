362 inline double Surface::getMinDistance(Point* point, double angle,
363  Point* intersection) {
364 
365  /* Point array for intersections with this Surface */
366  Point intersections[2];
367 
368  /* Find the intersection Point(s) */
369  int num_inters = this->intersection(point, angle, intersections);
370  double distance = INFINITY;
371 
372  /* If there is one intersection Point */
373  if (num_inters == 1) {
374  distance = intersections[0].distanceToPoint(point);
375  intersection->setX(intersections[0].getX());
376  intersection->setY(intersections[0].getY());
377  }
378 
379  /* If there are two intersection Points */
380  else if (num_inters == 2) {
381  double dist1 = intersections[0].distanceToPoint(point);
382  double dist2 = intersections[1].distanceToPoint(point);
383 
384  /* Determine which intersection Point is nearest */
385  if (dist1 < dist2) {
386  distance = dist1;
387  intersection->setX(intersections[0].getX());
388  intersection->setY(intersections[0].getY());
389  }
390  else {
391  distance = dist2;
392  intersection->setX(intersections[1].getX());
393  intersection->setY(intersections[1].getY());
394  }
395  }
396 
397  return distance;
398 }
